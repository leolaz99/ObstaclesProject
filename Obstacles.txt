using System;

namespace Game
{
    public struct Position
    {
        public int x;
        public int y;
    }

    public enum Effect { isSlowed, isStunned, isBlinded }
	

    public interface IObstacle    
    {
        string Action();            //descrive l'azione dell'ostacolo
    }

    class Obstacle
    {
        public float damage;        //descrive il danno dell'ostacolo al giocatore
        public Effect effect;       //descrive il debuff del giocatore per colpa dell'ostacolo
        public Position position;
    }

    
    class Bomb : Obstacle, IObstacle
    {
        public Bomb()
        {
            damage = 70f;
            effect=Effect.isSlowed;
        }

        public string Action()
        {
            return "Explode";
        }
    }

    class Smoke : Obstacle, IObstacle
    {
        public Smoke()
        {
            damage = 0f;
            effect=Effect.isBlinded;
        }

        public string Action()
        {
            return "Make fog";
        }
    }

    class Wind : Obstacle, IObstacle
    {
        public Wind()
        {
            damage = 0f;
            effect=Effect.isSlowed;
        }

        public string Action()
        {
            return "Make air";
        }
    }

    class Trapdoor : Obstacle, IObstacle
    {
        public Trapdoor()
        {
            damage = 10f;
            effect=Effect.isStunned;
        }

        public string Action()
        {
            return "Open door";
        }
    }

    class BearTrap : Obstacle, IObstacle
    {
        public BearTrap()
        {
            damage = 50f;
            effect=Effect.isStunned;
        }

        public string Action()
        {
            return "Close trap";
        }
    }
}

{
	public static void Main()
	{
	}
}
